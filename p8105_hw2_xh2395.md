pols_data = 
  read_csv("./data/pols-month.csv") %>% 
  janitor::clean_names() %>%
  separate(mon, into = c("year", "month", "day"), sep = "-") %>% 
  mutate(year = as.integer(year), month = as.integer(month), day = as.integer(day)) %>%

  select(-prez_gop, -prez_dem, -day)
